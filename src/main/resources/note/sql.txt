1.创建数据库
CREATE DATABASE IF NOT EXISTS yyz DEFAULT CHARSET utf8 COLLATE utf8_general_ci; 
2.创建User表，用于测试
CREATE TABLE `User` (
  `id` bigint(20) NOT NULL AUTO_INCREMENT COMMENT '自增主键',
  `loginIdAsOrgUser` bigint(15) DEFAULT NULL,
  `loginPhone` varchar(50) DEFAULT NULL COMMENT '登陆电话:只允许手机',
  `role` smallint(4) DEFAULT NULL COMMENT '用户角色',
  `orgUserId` bigint(20) DEFAULT NULL COMMENT '用户机构Id',
  `loginName` varchar(50) DEFAULT NULL COMMENT '登陆名称',
  `loginMail` varchar(50) DEFAULT NULL COMMENT '登陆邮箱',
  `loginNameBindTime` int(11) DEFAULT NULL COMMENT '登陆名称绑定时间',
  `loginMailBindTime` int(11) DEFAULT NULL COMMENT '登陆邮箱绑定时间',
  `loginPhoneBindTime` int(11) DEFAULT NULL COMMENT '登陆电话绑定时间',
  `loginPassword` varchar(50) NOT NULL COMMENT '登陆密码',
  `loginPasswordVisible` varchar(20) DEFAULT NULL COMMENT '明文密码',
  `loginLastType` tinyint(2) DEFAULT NULL COMMENT '最后一次登陆方式:名称,邮箱,手机',
  `loginLastTime` int(11) DEFAULT NULL COMMENT '最后一次登陆时间',
  `loginLastIp` varchar(40) DEFAULT NULL COMMENT '最后一次登录ip',
  `createIp` varchar(40) DEFAULT NULL COMMENT '注册ip',
  `status` tinyint(4) NOT NULL COMMENT '状态:默认0',
  `identityId` bigint(20) DEFAULT NULL COMMENT '实名信息编号',
  `rateSchemeId` bigint(20) DEFAULT NULL COMMENT '用户匹配的费率方案ID',
  `payPassword` varchar(50) DEFAULT NULL COMMENT '支付密码',
  `loginFailedTimes` tinyint(2) DEFAULT '0' COMMENT '登录失败次数',
  `parentId` bigint(20) DEFAULT NULL COMMENT '父用户id',
  `createTime` int(11) NOT NULL COMMENT '创建时间',
  `updateTime` int(11) NOT NULL COMMENT '更新时间',
  `deleteTime` int(11) DEFAULT NULL COMMENT '删除时间',
  `openPassword` varchar(32) DEFAULT NULL,
  `openToken` varchar(32) DEFAULT NULL,
  `userRemark` varchar(200) DEFAULT NULL COMMENT '用户说明',
  `reserve0` bigint(20) DEFAULT NULL COMMENT '备用字段',
  `reserve1` bigint(20) DEFAULT NULL COMMENT '备用字段',
  `reserve2` varchar(255) DEFAULT NULL COMMENT '备用字段',
  `loginLockExpireTime` int(11) DEFAULT NULL COMMENT '登录锁定超时时间',
  `userInvestmentAssessmentInfoId` bigint(20) DEFAULT NULL COMMENT '用户投资评估表有效ID',
  `groupId` bigint(20) DEFAULT NULL COMMENT '分组ID',
  `visibleInOrg` int(11) NOT NULL DEFAULT '1' COMMENT '隐藏显示:0代表隐藏，1代表显示',
  PRIMARY KEY (`id`),
  UNIQUE KEY `idx_loginIdAsOrgUser` (`loginIdAsOrgUser`) USING BTREE,
  UNIQUE KEY `idx_orgUserId_loginPhone ` (`orgUserId`,`loginPhone`) USING BTREE,
  UNIQUE KEY `idx_orgUserId_loginName ` (`orgUserId`,`loginName`) USING BTREE,
  KEY `idx_createTime ` (`createTime`),
  KEY `idx_updateTime ` (`updateTime`),
  KEY `idx_identityId ` (`identityId`),
  KEY `idx_orgUserId_role_groupId` (`orgUserId`,`role`,`groupId`)
) ENGINE=InnoDB AUTO_INCREMENT=0 DEFAULT CHARSET=utf8 COMMENT='用户表';
